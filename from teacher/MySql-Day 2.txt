Multiple table handling
=======================

emp,dept

select e.ename,e.deptno,d.deptno,d.dname
from emp e, dept d 
where e.deptno=d.deptno;   

emp,dept,salgrade
select e.ename,e.deptno,d.deptno,d.dname,e.sal,s.grade
from emp e, dept d, salgrade s
where e.deptno=d.deptno and (e.sal between s.losal and s.hisal);

regular expressions
===================

select * 
from emp
where REGEXP_LIKE(column_name,'pattern');

^ : first letter
$ : last letter
. : single letter
.* : multiple letters
[0-9] : digits \d
[a-z]:letters

-list all emp starting with letter A
select * 
from emp
where REGEXP_LIKE(ename,'^A');

-list all emp where  letter N comes in between
select * 
from emp
where REGEXP_LIKE(ename,'.*N.*');

-list all emp where  letter N at the end
select * 
from emp
where REGEXP_LIKE(ename,'.*N$');

-list all emp strting with MSA  
select * 
from emp
where REGEXP_LIKE(ename,'^[MSA].*');

-not starting with A
select * 
from emp
where REGEXP_LIKE(ename,'^[^A].*');

-name starts with A and ends with n
select * 
from emp
where REGEXP_LIKE(ename,'^A.*N$');

-name starts with A and ends with n or starts with m and ends with n or
starts with s and ends with h
select * 
from emp
where REGEXP_LIKE(ename,'^A.*N$|^m.*n$|^s.*h$');

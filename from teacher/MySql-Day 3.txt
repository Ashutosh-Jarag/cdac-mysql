product(pid,pname,cid) dependent table
        ---       ---
         pk       fk

create table cat
(
  cat_id int,
  cname varchar(15) unique not null,
  constraint pk_catid primary key(cat_id)
);

insert into cat
values(10,'Food');
insert into cat
values(20,'Clothes');
insert into cat
values(30,'Stationary');

create table product
(pid int,
 pname varchar(30) not null,
 cid int,
 constraint pk_pid primary key(pid),
 constraint fk_cid foreign key(cid) references cat(cat_id) 
);

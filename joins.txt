--2
select p.pname, s.sid, s.sname
from product p
left join salesman s
on p.sid = s.sid
where s.sname = 'kirti';


--3

select e.empno, e.ename, e.sal, d.deptno, e.sal
from emp e
join dept d
on e.deptno = d.deptno
join salgrade s
on e.sal between s.losal and s.hisal
where s.grade = 2;


--4
select e.empno, e.ename, e.deptno, d.dname, e.hiredate, d.loc
from emp e
left join dept d
on e.deptno = d.deptno
where e.job = 'manager' and year(e.hiredate) = '1981';

from product p

on p.cid = c.cid
where c.cname = 'chips'; 

--10
select e.empno, e.ename, e.deptno, d.dname, e.sal, s.grade, e.hiredate
from emp e
join dept d
on e.deptno = d.deptno
join salgrade s
on e.sal between s.losal and s.hisal
where year(e.hiredate) = 1981 or year(e.hiredate) = 1980
order by e.hiredate;

delimiter //
create procedure perfect_value(IN n INT, OUT result INT)
BEGIN 
	DECLARE i INT DEFAULT 1;
	DECLARE total int DEFAULT 0;
	
	while i < n DO
		IF n%i = 0 then 
			SET total = total +i;
		END IF;
		SET i = i+1;
	END WHILE;
	
	IF total = n THEN
		SET result = 1;
	ELSE 
		SET result = 0;
	END IF;
end // 
delimiter ; 


call perfect_value(12, @a);

select @a 'result';

create procedure factorial(in n int, out result int)
begin 
	declare i int default 1;
	declare fact int default 1;
	
	while i <= n do
		set fact = fact *i;
		set i = i+1;
	end while;
	
	set result = fact;
end // 
delimiter ;

call factorial(5,@result);
select @result;

delimiter //
create procedure print_seq1(in m int, in n int)
begin
	declare start int default 0;
	declare end_val int default 0;
	
	if m<n then
		set start = m;
		set end_val = n;
	else
		set start = n;
		set end_val = m;
	end if;
	
	while start <= end_val do
		select start as number; 
		set start = start +1;
	end while;
end // 
delimiter ;

		
CALL print_seq1(3, 7); 
CALL print_seq1(7, 3);

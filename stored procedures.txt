CREATE PROCEDURE p5(IN n INT)
BEGIN
    IF n IS NOT NULL THEN
        SELECT empno, ename, 'Found' AS message 
        FROM emp
        WHERE empno = n;
    ELSE
        SELECT 'Employee Not Found' AS message;
    END IF;
END //
DELIMITER ;

CREATE PROCEDURE p51(IN n INT, OUT result INT)
BEGIN
	select count(*)as total
	from emp 
	where empno = n;
	
	IF total > 0 then
		select ename, sal, 'FOUND' as message from emp 
		where empno = n;
		set result = 'FOUND';
	ELSE
		SELECT emame, sal, 'NOT FOUND' as message;
		set result = 'NOT FOUND';
	END IF;
END // 
DELIMITER ;

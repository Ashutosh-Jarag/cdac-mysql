46. Find customers whose **account balance is greater than the average balance 
    of their branch**.
	
SELECT name, balance, branch_name
FROM (
    SELECT c.name, a.balance, b.branch_name,
    AVG(a.balance) OVER(PARTITION BY c.branch_id) AS branch_avg
    FROM customer c
    JOIN account a ON c.customer_id = a.customer_id
    JOIN branch b ON c.branch_id = b.branch_id
) AS temp
WHERE balance > branch_avg;

select sum(sal) as sum_sal 
from emp
group by deptno;


--2
select deptno,sum(sal) 
from emp
group by deptno
having sum(sal) > 10000;

--3
SELECT job, SUM(sal) AS total_salary
FROM emp
WHERE deptno != 30
GROUP BY job
HAVING SUM(sal) > 5000;


--4
select job, max(sal) as salary
from emp
where job = 'manager' #or job = 'clerk' or job = 'salesman' and (deptno = 10 or deptno= 20)
group by job
order by max(sal) desc;


select job, max(sal) as salary
from emp
where job in ('manager','clerk','salesman') and  deptno in(10,20)
group by job
order by max(sal) desc;

--6
